# Some simple cases 

   > "hello world";
   > 5 == 3;

   > 1 + true;
   > Type error: Binary expression at Unknown has non-int right operand

   > var a = "cat";
   > a + 123;
   Type error: Binary expression at Unknown has non-int left operand

# If expressions

   > if 5 == 5 { 3 } else { 2 } == 3;

   > if 1 { true } else { false };
   Type error: If expression at Unknown  has non-bool condition

   > if true { 123 } else { true };
   Type error: If expression at Unknown  has arms of different types

# Function declarations

   > fun f(i: Int) String {
       return i;
     }
   Type error: Return type does not match the function declaration

   > fun f(i: Int) Int {
       return i;                   # <<<---- the function block as a whole returns Unit
     }
   Type error: Return type does not match the function declaration

   > fun f(i: Int) Unit {
       if i { 123; } else { 123; };
     }
   # While typechecking the body...
   Type error: If expression at Unknown  has non-bool condition

   > fun f(i: Int) Int {
       i
     }

   # Ok!

# Function calls

   > f(true);
   Type error: Function f at Unknown  and its invocation at Line = 16, column = 1 do not correspond

   > f(123);

   # Ok!
